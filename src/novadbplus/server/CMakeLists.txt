add_executable(novadbplus main.cpp)
if(NOT WITH_ASAN AND NOT WITH_TSAN)
    if (BUILD_novadb_STATIC)
        add_executable(novadbplus_static main.cpp)
    endif()
endif()
if(CMAKE_COMPILER_IS_GNUCC)
    # command uses global variables to self-regist, linking flag whole-archive is required.
    target_link_libraries(novadbplus -Wl,--whole-archive commands -Wl,--no-whole-archive)
    target_link_libraries(novadbplus server_params server ${SYS_LIBS})
    if(NOT WITH_ASAN AND NOT WITH_TSAN)
        set_target_properties(novadbplus PROPERTIES LINK_FLAGS "-static-libgcc -static-libstdc++")
        if (BUILD_novadb_STATIC)
            target_link_libraries(novadbplus_static -Wl,--whole-archive commands -Wl,--no-whole-archive)
            target_link_libraries(novadbplus_static server_params server ${SYS_LIBS})
            set_target_properties(novadbplus_static PROPERTIES LINK_FLAGS "-static")
        endif()
    endif()

else()
    target_link_libraries(novadbplus commands server_params server ${SYS_LIBS})
    set_target_properties(novadbplus PROPERTIES LINK_FLAGS "/WHOLEARCHIVE:commands")
    if (BUILD_novadb_STATIC)
        target_link_libraries(novadbplus_static commands server_params server ${SYS_LIBS})
        set_target_properties(novadbplus_static PROPERTIES LINK_FLAGS "/WHOLEARCHIVE:commands")
    endif()
endif()
